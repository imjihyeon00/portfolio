$('.paraDot ul li').click(function (e) {
    e.preventDefault();
    let target = $(this);         //타겟 설정
    let index = target.index();   //인덱스 설정
    let section = $('.content-item').eq(index);
    let offset = section.offset().top;

    $('html, body').animate({ scrollTop: offset }, 700, 'easeInBack');

});

$(window).scroll(function () {
    let scrollTop = $(this).scrollTop();
    $('.paraScroll span').text(parseInt(scrollTop));

    $('.content-item').each((idx, item) => {
        if (scrollTop >= $('.content-item').eq(idx).offset().top - 2) {
            $('.paraDot ul li').eq(idx).addClass('active').siblings().removeClass('active');
        }
    });
});
