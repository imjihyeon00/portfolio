document.querySelectorAll('.paraNav ul li a').forEach((el, idx) => {
    el.addEventListener('click', function (e) {
        e.preventDefault();
        document.querySelector(el.getAttribute('href')).scrollIntoView({
            behavior: 'smooth'
        });
    });
});

window.addEventListener('scroll', function () {
    let scrollTop = document.documentElement.scrollTop || window.scrollY || window.pageYOffset; 

    document.querySelectorAll('.paraNav ul li').forEach((el, idx) => {
        if (scrollTop >= document.getElementById('section' + (idx + 1)).offsetTop - 2) {
            document.querySelectorAll('.paraNav ul li').forEach(el => el.classList.remove('active'));
            document.querySelector('.paraNav ul li:nth-child(' + (idx + 1) + ')').classList.add('active');
        }
    });
});
